## üåô Advanced Dark Mode Setup

### Dark Mode JavaScript

**File: `resources/js/darkmode.js`**
```javascript
class DarkModeManager {
    constructor() {
        this.init();
        this.bindEvents();
    }
    
    init() {
        // Initialize dark mode based on preference
        if (this.getStoredTheme() === 'dark' || 
            (!this.getStoredTheme() && this.getSystemPreference() === 'dark')) {
            this.enableDarkMode();
        } else {
            this.disableDarkMode();
        }
        
        // Update toggle button state
        this.updateToggleButton();
    }
    
    bindEvents() {
        // Listen for system theme changes
        window.matchMedia('(prefers-color-scheme: dark)')
            .addEventListener('change', (e) => {
                if (!this.getStoredTheme()) {
                    if (e.matches) {
                        this.enableDarkMode();
                    } else {
                        this.disableDarkMode();
                    }
                    this.updateToggleButton();
                }
            });
    }
    
    toggle() {
        if (this.isDarkMode()) {
            this.disableDarkMode();
            this.setStoredTheme('light');
        } else {
            this.enableDarkMode();
            this.setStoredTheme('dark');
        }
        this.updateToggleButton();
        
        // Dispatch custom event
        window.dispatchEvent(new CustomEvent('themeChanged', {
            detail: { theme: this.isDarkMode() ? 'dark' : 'light' }
        }));
    }
    
    enableDarkMode() {
        document.documentElement.classList.add('dark');
        this.updateMetaTheme('#1f2937'); // gray-800
    }
    
    disableDarkMode() {
        document.documentElement.classList.remove('dark');
        this.updateMetaTheme('#ffffff');
    }
    
    isDarkMode() {
        return document.documentElement.classList.contains('dark');
    }
    
    getStoredTheme() {
        return localStorage.getItem('theme');
    }
    
    setStoredTheme(theme) {
        localStorage.setItem('theme', theme);
    }
    
    getSystemPreference() {
        return window.matchMedia('(prefers-color-scheme: dark)').matches ? 'dark' : 'light';
    }
    
    updateToggleButton() {
        const toggleButtons = document.querySelectorAll('[data-theme-toggle]');
        toggleButtons.forEach(button => {
            const sunIcon = button.querySelector('.theme-icon-sun');
            const moonIcon = button.querySelector('.theme-icon-moon');
            
            if (this.isDarkMode()) {
                sunIcon?.classList.remove('hidden');
                moonIcon?.classList.add('hidden');
            } else {
                sunIcon?.classList.add('hidden');
                moonIcon?.classList.remove('hidden');
            }
        });
    }
    
    updateMetaTheme(color) {
        let metaTheme = document.querySelector('meta[name="theme-color"]');
        if (!metaTheme) {
            metaTheme = document.createElement('meta');
            metaTheme.name = 'theme-color';
            document.head.appendChild(metaTheme);
        }
        metaTheme.content = color;
    }
}

// Initialize dark mode manager
const darkMode = new DarkModeManager();

// Export for global access
window.darkMode = darkMode;
window.toggleDarkMode = () => darkMode.toggle();

// Auto-save form data on theme change (optional)
window.addEventListener('themeChanged', (e) => {
    console.log('Theme changed to:', e.detail.theme);
    // You can add analytics tracking here
});
```

```javascript
class SidebarManager {
    constructor() {
        this.sidebar = document.querySelector('[data-sidebar]');
        this.overlay = document.querySelector('[data-sidebar-overlay]');
        this.toggleButtons = document.querySelectorAll('[data-sidebar-toggle]');
        this.isOpen = false;
        
        if (this.sidebar) {
            this.init();
        }
    }
    
    init() {
        this.bindEvents();
        this.handleResize();
    }
    
    bindEvents() {
        // Toggle buttons
        this.toggleButtons.forEach(button => {
            button.addEventListener('click', () => this.toggle());
        });
        
        // Overlay click
        if (this.overlay) {
            this.overlay.addEventListener('click', () => this.close());
        }
        
        // Escape key
        document.addEventListener('keydown', (e) => {
            if (e.key === 'Escape' && this.isOpen) {
                this.close();
            }
        });
        
        // Resize handler
        window.addEventListener('resize', () => this.handleResize());
        
        // Close on navigation (mobile)
        document.addEventListener('click', (e) => {
            if (window.innerWidth < 768 && 
                e.target.closest('[data-sidebar-link]') && 
                this.isOpen) {
                this.close();
            }
        });
    }
    
    toggle() {
        if (this.isOpen) {
            this.close();
        } else {
            this.open();
        }
    }
    
    open() {
        this.sidebar.classList.add('open');
        this.overlay?.classList.remove('hidden');
        document.body.classList.add('overflow-hidden', 'lg:overflow-auto');
        this.isOpen = true;
        
        // Focus trap for accessibility
        this.focusTrap();
    }
    
    close() {
        this.sidebar.classList.remove('open');
        this.overlay?.classList.add('hidden');
        document.body.classList.remove('overflow-hidden');
        this.isOpen = false;
    }
    
    handleResize() {
        if (window.innerWidth >= 1024) { // lg breakpoint
            this.close();
        }
    }
    
    focusTrap() {
        const focusableElements = this.sidebar.querySelectorAll(
            'a[href], button, textarea, input, select, [tabindex]:not([tabindex="-1"])'
```

```javascript
    // Confirm dialog
    static async confirm(message, title = 'Confirm') {
        return new Promise((resolve) => {
            const modal = document.createElement('div');
            modal.className = 'fixed inset-0 z-50 flex items-center justify-center p-4 bg-gray-500 bg-opacity-75';
            modal.innerHTML = `
                <div class="bg-white dark:bg-gray-800 rounded-lg shadow-xl max-w-md w-full p-6">

                        <button data-cancel class="btn btn-secondary">Cancel</button>
                        <button data-confirm class="btn btn-danger">Confirm</button>
                    </div>
                </div>
            `;
            
            document.body.appendChild(modal);
            
            modal.querySelector('[data-cancel]').addEventListener('click', () => {
                document.body.removeChild(modal);
                resolve(false);
            });
            
            modal.querySelector('[data-confirm]').addEventListener('click', () => {
                document.body.removeChild(modal);
                resolve(true);
            });
            
            // ESC key
            const handleEsc = (e) => {
                if (e.key === 'Escape') {
                    document.body.removeChild(modal);
                    document.removeEventListener('keydown', handleEsc);
                    resolve(false);
                }
            };
            document.addEventListener('keydown', handleEsc);
        });
    }
    
    // Copy to clipboard
    static async copyToClipboard(text) {
        try {
            await navigator.clipboard.writeText(text);
            this.showToast('Copied to clipboard!');
        } catch (err) {
            // Fallback for older browsers
            const textArea = document.createElement('textarea');
            textArea.value = text;
            document.body.appendChild(textArea);
            textArea.select();
            document.execCommand('copy');
            document.body.removeChild(textArea);
            this.showToast('Copied to clipboard!');
        }
    }
    
    // Format file size
    static formatFileSize(bytes) {
        if (bytes === 0) return '0 Bytes';
        const k = 1024;
        const sizes = ['Bytes', 'KB', 'MB', 'GB'];
        const i = Math.floor(Math.log(bytes) / Math.log(k));
        return parseFloat((bytes / Math.pow(k, i)).toFixed(2)) + ' ' + sizes[i];
    }
    
    // Debounce function
    static debounce(func, wait) {
        let timeout;
        return function executedFunction(...args) {
            const later = () => {
                clearTimeout(timeout);
                func(...args);
            };
            clearTimeout(timeout);
            timeout = setTimeout(later, wait);
        };
    }

        const debouncedSave = AdminUtils.debounce(() => {

            const key = `autosave_${form.dataset.autoSave}`;
            const data = Object.fromEntries(formData);
            localStorage.setItem(key, JSON.stringify(data));
        }, 1000);
        
        inputs.forEach(input => {
            input.addEventListener('input', debouncedSave);
        });
        
        // Restore saved data on page load

        if (savedData) {

                    if (input) input.value = value;
                });
            } catch (e) {

            }
        }
```

```php
<?php

namespace Tests\Feature;

use App\Models\User;
use Illuminate\Foundation\Testing\RefreshDatabase;
use Tests\TestCase;

class AdminDashboardTest extends TestCase
{
    use RefreshDatabase;

    protected function setUp(): void
    {
        parent::setUp();
        
        // Create admin user
        $this->admin = User::factory()->create([
            'email' => 'admin@example.com',
            'is_admin' => true,
        ]);
    }

    public function test_admin_can_access_dashboard()
    {
        $response = $this->actingAs($this->admin)
                         ->get('/admin/dashboard');

        $response->assertStatus(200)
                 ->assertViewIs('admin.dashboard')
                 ->assertViewHas(['stats', 'recentUsers']);
    }

    public function test_regular_user_cannot_access_admin()
    {
        $user = User::factory()->create(['is_admin' => false]);

        $response = $this->actingAs($user)
                         ->get('/admin/dashboard');

        $response->assertStatus(403);
    }

    {
        $response = $this->get('/admin/dashboard');

        $response->assertRedirect('/login');
    }
```

### Performance Optimization

1. **CSS Optimization**
```css
/* Use @apply sparingly in production */
.btn-primary {
    @apply bg-blue-600 text-white hover:bg-blue-700; /* OK */
}

/* Prefer utility classes in HTML */
<button class="bg-blue-600 text-white hover:bg-blue-700 px-4 py-2 rounded">
```

2. **JavaScript Optimization**
```javascript
// Lazy load components
const modal = () => import('./components/modal.js');

// Debounce search inputs
const debouncedSearch = AdminUtils.debounce(handleSearch, 300);
```

3. **Laravel Optimization**
```php
// Use database indexes
Schema::table('users', function (Blueprint $table) {
    $table->index(['created_at', 'is_active']);
});

// Eager loading
$users = User::with(['roles', 'profile'])->paginate(20);
```

### Security Best Practices

1. **CSRF Protection**
```php
// Always use CSRF tokens
<form method="POST" action="{{ route('admin.users.store') }}">
    @csrf
    <!-- form fields -->
</form>
```

2. **Authorization**
```php
// Use policies for authorization
public function view(User $user, Post $post)
{
    return $user->isAdmin() || $user->id === $post->user_id;
}
```

3. **Input Validation**
```php
public function store(Request $request)
{
    $validated = $request->validate([
        'name' => 'required|string|max:255',
        'email' => 'required|email|unique:users',
        'password' => 'required|min:8|confirmed',
    ]);
    
    // Process validated data
}
```
